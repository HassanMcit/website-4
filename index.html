Sorting Element In Array
      
                  
                    int i, j, Temp, n, number[30];
                    for (i = 0; i < n; ++i)
                    {
                      for (j = i + 1; j < n; ++j)
                        {
                          if (number[i] > number[j])
                          {
                            Temp =  number[i];
                            number[i] = number[j];
                            number[j] = Temp;
                          }
                        }
                      }
                    }
  
                    printf("The numbers arranged in ascending order are given below \n");
                    for (i = 0; i < n; ++i)
                    {
                      printf("%d\n", number[i]);
                    }
